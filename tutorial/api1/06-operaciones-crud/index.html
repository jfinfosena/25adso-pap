    Operaciones CRUD

    Introducci贸n

    CRUD es un acr贸nimo que representa las cuatro operaciones b谩sicas de persistencia de datos:

    • Create (Crear)
    • Read (Leer)
    • Update (Actualizar)
    • Delete (Eliminar)

    En FastAPI con SQLAlchemy, estas operaciones se implementan como funciones que interact煤an con la base de datos usando sesiones de SQLAlchemy.

    驴Por qu茅 separar las operaciones CRUD?

    Ventajas de la separaci贸n

    1. Reutilizaci贸n: Las mismas operaciones pueden usarse en diferentes endpoints
    2. Testabilidad: F谩cil de testear independientemente
    3. Mantenimiento: L贸gica de base de datos centralizada
    4. Consistencia: Operaciones estandarizadas

from pydantic import BaseModel
from sqlalchemy.orm import Session
from app.database.base import BaseModel as DBBaseModel

# Tipos gen茅ricos
ModelType = TypeVar("ModelType", bound=DBBaseModel)
CreateSchemaType = TypeVar("CreateSchemaType", bound=BaseModel)
UpdateSchemaType = TypeVar("UpdateSchemaType", bound=BaseModel)

class CRUDBase(Generic[ModelType, CreateSchemaType, UpdateSchemaType]):
    """
    Clase base para operaciones CRUD.

    Proporciona operaciones b谩sicas que pueden ser heredadas
    y extendidas por clases CRUD espec铆ficas.
    """

    def __init__(self, model: Type[ModelType]):
        """
        Inicializar con el modelo SQLAlchemy.

        Args:
            model: Clase del modelo SQLAlchemy
        """
        self.model = model

    def get(self, db: Session, id: Any) -> Optional[ModelType]:
        """
        Obtener un registro por ID.

        Args:
            db: Sesi贸n de base de datos
            id: ID del registro

        Returns:
            Registro encontrado o None
        """
        return db.query(self.model).filter(self.model.id == id).first()

    def get_multi(
        self, 
        db: Session, 
        *, 
        skip: int = 0, 
        limit: int = 100
    ) -> List[ModelType]:
        """
        Obtener m煤ltiples registros con paginaci贸n.

        Args:
            db: Sesi贸n de base de datos
            skip: N煤mero de registros a saltar
            limit: N煤mero m谩ximo de registros a retornar

        Returns:
            Lista de registros
        """
        return db.query(self.model).offset(skip).limit(limit).all()

    def create(self, db: Session, *, obj_in: CreateSchemaType) -> ModelType:
        """
        Crear un nuevo registro.

        Args:
            db: Sesi贸n de base de datos
            obj_in: Datos para crear el registro

        Returns:
            Registro creado
        """
        obj_in_data = jsonable_encoder(obj_in)
        db_obj = self.model(**obj_in_data)
        db.add(db_obj)
        db.commit()
        db.refresh(db_obj)
        return db_obj

    def update(
        self,
        db: Session,
        *,
        db_obj: ModelType,
        obj_in: Union[UpdateSchemaType, Dict[str, Any]]
    ) -> ModelType:
        """
        Actualizar un registro existente.

        Args:
            db: Sesi贸n de base de datos
            db_obj: Registro existente en la base de datos
            obj_in: Datos para actualizar

        Returns:
            Registro actualizado
        """
        obj_data = jsonable_encoder(db_obj)

        if isinstance(obj_in, dict):
            update_data = obj_in
        else:
            update_data = obj_in.dict(exclude_unset=True)

        for field in obj_data:
            if field in update_data:
                setattr(db_obj, field, update_data[field])

        db.add(db_obj)
        db.commit()
        db.refresh(db_obj)
        return db_obj

    def remove(self, db: Session, *, id: int) -> ModelType:
        """
        Eliminar un registro por ID.

        Args:
            db: Sesi贸n de base de datos
            id: ID del registro a eliminar

        Returns:
            Registro eliminado
        """
        obj = db.query(self.model).get(id)
        db.delete(obj)
        db.commit()
        return obj

    def count(self, db: Session) -> int:
        """
        Contar el total de registros.

        Args:
            db: Sesi贸n de base de datos

        Returns:
            N煤mero total de registros
        """
        return db.query(self.model).count()

    def exists(self, db: Session, id: int) -> bool:
        """
        Verificar si existe un registro con el ID dado.

        Args:
            db: Sesi贸n de base de datos
            id: ID a verificar

        Returns:
            True si existe, False si no
        """
        return db.query(self.model).filter(self.model.id == id).first() is not None

class CRUDUser(CRUDBase[User, UserCreate, UserUpdate]):
    """

        Args:
            db: Sesi贸n de base de datos
            username: Nombre de usuario

        Returns:
            Usuario encontrado o None
        """

    def search_users(
        self, 
        db: Session, 
        *, 
        query: str, 
        skip: int = 0, 
        limit: int = 100
    ) -> List[User]:
        """
        Buscar usuarios por nombre de usuario, email o nombre completo.

        Args:
            db: Sesi贸n de base de datos

                User.email.ilike(search_term),
                User.full_name.ilike(search_term)
            )
        ).offset(skip).limit(limit).all()

    def get_active_users(
        self, 
        db: Session, 
        *, 
        skip: int = 0, 
        limit: int = 100
    ) -> List[User]:

        Obtener usuarios activos.

        Args:
            db: Sesi贸n de base de datos
            skip: Registros a saltar
            limit: L铆mite de registros

        Returns:
            Lista de usuarios activos
        """
        return db.query(User).filter(
            User.is_active == True
        ).offset(skip).limit(limit).all()

    def create_user(self, db: Session, *, user: UserCreate) -> User:
        """
        Crear un nuevo usuario con validaciones adicionales.

        Args:
            db: Sesi贸n de base de datos
            user: Datos del usuario a crear

        Returns:
            Usuario creado

            ValueError: Si el email o username ya existen

            raise ValueError("El email ya est谩 registrado")

        # Verificar si el username ya existe
        if self.get_by_username(db, username=user.username):
            raise ValueError("El nombre de usuario ya est谩 registrado")

        return self.create(db, obj_in=user)

    def update_user(
        self, 
        db: Session, 
        *, 
        db_obj: User, 
        obj_in: UserUpdate
    ) -> User:
        """
        Actualizar usuario con validaciones adicionales.

        Args:
            db: Sesi贸n de base de datos
            db_obj: Usuario existente
            obj_in: Datos de actualizaci贸n

        Returns:
            Usuario actualizado

        Raises:
            ValueError: Si el nuevo email o username ya existen
        """

        if obj_in.email and obj_in.email != db_obj.email:
            existing_user = self.get_by_email(db, email=obj_in.email)
            if existing_user and existing_user.id != db_obj.id:
                raise ValueError("El email ya est谩 registrado")

        # Verificar username si se est谩 actualizando
        if obj_in.username and obj_in.username != db_obj.username:
            existing_user = self.get_by_username(db, username=obj_in.username)
            if existing_user and existing_user.id != db_obj.id:
                raise ValueError("El nombre de usuario ya est谩 registrado")

        return self.update(db, db_obj=db_obj, obj_in=obj_in)

    def deactivate_user(self, db: Session, *, user_id: int) -> User:
        """
        Desactivar un usuario (soft delete).

        Args:
            db: Sesi贸n de base de datos
            user_id: ID del usuario a desactivar

        Returns:
            Usuario desactivado
        """
        user = self.get(db, id=user_id)
        if user:
            user.is_active = False
            db.commit()
            db.refresh(user)
        return user
